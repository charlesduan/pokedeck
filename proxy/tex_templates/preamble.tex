\def\cardbox#1{%
    \leavevmode
    \vbox{%
        \hrule
        \hbox{%
            \vrule
                \vbox to 8.8cm{%
                    \hrule width 6.3cm height 0pt
                    \hsize=6.3cm
                    #1%
                }%
            \vrule
        }%
        \hrule
    }\allowbreak\ignorespaces
}

\def\blankcard{\cardbox{\vfill}}%

\long\def\summarycard#1#2#3{%
    \cardbox{%
        \begin{center}\textbf{#1}\end{center}
        \begingroup
            \parindent=1em
            #2\par
        \endgroup
        \vskip 6pt
        \hbox
        \setbox0=\vbox{\divide\hsize by 2 \advance\hsize -1.5pt #3}%
