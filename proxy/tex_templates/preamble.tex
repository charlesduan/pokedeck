\long\def\cardbox#1{%
    \leavevmode
    \vbox{%
        \hrule
        \hbox{%
            \vrule
                \vbox to 8.8cm{%
                    \hrule width 6.3cm height 0pt
                    \hsize=6.3cm
                    #1%
                }%
            \vrule
        }%
        \hrule
    }\allowbreak\ignorespaces
}

\def\blankcard{\cardbox{\vfill}}%

\long\def\summarycard#1#2#3{%
    \cardbox{%
        \begin{center}\textbf{#1}\end{center}
        \begingroup
            \parindent=1em
            #2\par
        \endgroup
        \vskip 6pt
        #3%
        \vfill
    }%
}
\def\summaryline#1#2#3#4{%
    \leavevmode
    \begingroup
        \rightskip=3em
        \parfillskip=0pt
        #1 #2\unskip\nobreak\hskip 6pt plus 1fill\relax
        \hbox{#3 #4}\nobreak\hskip-\rightskip\hbox{}%
        \par
    \endgroup
}
